const searchcontainer = document.querySelector('.search-container')
const formsearch = document.querySelector('.search')
const inputsearch = document.querySelector('#search-input')

function showsearch(infor) {
    const titlespecialization = document.querySelector('title-specialization')
    const titlehospital = document.querySelector('title-hospital')
    const titledoctor = document.querySelector('title-doctor')
    infor.forEach(data => {
        const spanhospital = document.createElement('span')
        const spanspecial = document.createElement('span') 
        const spandoctor = document.createElement('span')
        spandoctor.innerText = data.first_name + data.last_name
        spanspecial.innerText = data.specialization_name
        spanhospital.innerText = data.hospitalname
        titlespecialization.appendChild(spanspecial)
        titlehospital.appendChild(spanhospital)
        titledoctor.appendChild(spandoctor)
    })
}

async function getsearchresult(input) {
     const datajson = await fetch('')
     const data = await datajson.json()
     data.forEach(infor => {
         if(infor.join('').contains(input.trim())) {
             showsearch(infor)
         }
     })
    
}

formsearch.addEventListener('input', (e) => {
    e.preventDefault()
  
    getsearchresult(e.target.value)
})
